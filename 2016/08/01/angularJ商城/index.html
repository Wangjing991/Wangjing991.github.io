angularJ商城

简介

AngularJS使用了不同的方法，它尝试去补足HTML本身在构建应用方面的缺陷。AngularJS通过使用我们称为标识符(directives)的结构，让浏览器能够识别新的语法。

angularJs商城应用

此商城应用使用了ng-app定义了一个指令程序、angular.module全局对象、通过directive函数自定义指令、使用$http服务进行ajax网络请求、使用系统内置的过滤器：currency和uppercase等、DOM操作指令，比如ng-show和ng-hide、大量使用ng-repeat指令，用于循环减少重复代码、ng-click用于添加点击事件

给大家提供线上链接效果点击商城即可看到,要想下载源码的可以去点击Github下载

app.js模块也是总控制模块

// 模块（module）angularjs提供的全局函数去创建模块;
var app=angular.module("shop",["cFilters","ngRoute"]);

// 自定义指令
app.directive("cartInfo",function(){
    return {
        restrict:"CAEM", //限定使用场所C只显示class，A使用属性类型的,E使用标签类型的,M代表注释形式，默认支持AE(即不写restrict时候显示)
        // replace:true, //替换注释的内容显示
        transclude:true,
        templateUrl:"template/template.html" //自定义代码块路径
    }
})

app.controller("mainCtrl",function($scope,$http){ //定义控制器，两个参数，一参控制器名称，二参函数
    $scope.cartData=[]; //为了后面多个独立模块都能使用此参数在主模块里面设置变量
    $scope.count=0; //购物车数量
    $scope.total=0; //购物车总结
    $scope.getCartCount=function(){
        $scope.count=0;
        $scope.total=0;
        for(var i=0;i<$scope.cartData.length;i++){
            $scope.count+=$scope.cartData[i].count
            $scope.total+=$scope.cartData[i].count*$scope.cartData[i].price;
        }
        return $scope.count;
    }
});
    app.config(function($routeProvider){   //路由配置模块
        $routeProvider.when("/products",{templateUrl:"product.html"});
        $routeProvider.when("/checkout",{templateUrl:"checkout.html"});
        $routeProvider.otherwise("/products",{templateUrl:"product.html"});
    })

自定义过滤函数

    var filters = angular.module("cFilters",[]);
filters.filter("unique",function(){
    return function(data,propertyName){
        if (angular.isArray(data)) {
            var resultArray=[];
            for (var i=0;i<data.length;i++){
                var category=data[i][propertyName];
                if (resultArray.indexOf(category)<0) {
                    resultArray.push(category);
                }
            }
            return resultArray;
        }
        else
        {
            return [];
        }
    }
})
